slider = alt.binding_range(min=1955, max=2005, step=5, name='Cutoff ')
selector = alt.param(name='SelectorName', value=1970, bind=slider)
predicate = alt.datum.year == selector

alt.Chart(health_wealth).mark_circle().encode(
    alt.X('fertility:Q', scale = alt.Scale(zero=False)),
    alt.Y('life_expect:Q', scale = alt.Scale(zero=False)),
    size ='pop:Q',
    color = 'cluster:N',
    opacity=alt.when(predicate).then(alt.value(1)).otherwise(alt.value(0))
).add_params(selector)

slider = alt.binding_range(min=driving.year.min(), max=driving.year.max()+1, step=1, name='Cutoff ')
selector = alt.param(name='SelectorName', value=1960, bind=slider)
predicate = alt.datum.year < selector

points = alt.Chart(driving).mark_point().encode(
    alt.X('miles:Q', scale = alt.Scale(zero=False), title = 'Miles per person per year'),
    alt.Y('gas:Q', scale = alt.Scale(zero=False), title = 'Cost per gallon'),
)

line = alt.Chart(driving).mark_line().encode(
    alt.X('miles:Q', scale = alt.Scale(zero=False), title = 'Miles per person per year'),
    alt.Y('gas:Q', scale = alt.Scale(zero=False), title = 'Cost per gallon'),
    alt.Order('year'),
).add_params(selector).transform_filter(predicate)

text = alt.Chart(driving).mark_text(align = 'right').encode(
    alt.X('miles:Q', scale = alt.Scale(zero=False), title = 'Miles per person per year'),
    alt.Y('gas:Q', scale = alt.Scale(zero=False), title = 'Cost per gallon'),
    alt.Text('year')
).add_params(selector).transform_filter(predicate)

(points + line + text).properties(width = 500, height = 300)

single = alt.selection_point(on='mouseover', nearest = True)

base = alt.Chart(crashes).mark_circle().encode(
    alt.X(alt.repeat("column"), type='quantitative', scale=alt.Scale(zero=False)),
    alt.Y('state', sort=alt.EncodingSortField(field='rank', order='descending')),
    color=alt.condition(single, alt.value('purple'), alt.value('lightgray'))
).properties(width = 150,height = 400).add_params(single)

base.repeat(
    column=['total','speeding','alcohol']
